IN - MySQL условие IN используется, чтобы уменьшить необходимость использования нескольких условий OR в операторе SELECT, INSERT, UPDATE или DELETE.

SELECT *
  FROM contacts
 WHERE last_name IN ('Bernard', 'Boy', 'Tomas');

Команда AS задает новое имя полям или таблицам при выборке из базы (то есть в самой базе изменения не происходят).

Связывание таблиц
SELECT columns
 FROM table1
 INNER JOIN table2
 ON table1.column = table2.column;

SELECT suppliers.supplier_id, suppliers.supplier_name, orders.order_date
FROM suppliers
INNER JOIN orders
ON suppliers.supplier_id = orders.supplier_id;
