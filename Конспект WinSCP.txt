ВПС - виртуальная машина с множеством ОС - вариант для больших проектов

Выделеный сервер - отдельный комп, для супербольших проектов

SSL - шифрованное соединение для домена (сайт безопасен)

var - www - папка сайта

в настройках apache указать dns и домен

var - logs - apache2 - access.log - лог доступа

error.log  - ошибки сервера

РАБОТА С ГИТОМ

ssh - утилита для подкл к серверу

ssh user@ip -p 22 // к какому порту, если ftp то 22
ls -la // посмотреть файловую структуру
apt install <название программы> // установка утилит
htop // утилита для красивого отображения статистики сервера (как диспетчер задач)
vim /etc/apache2/sites-available/<название проекта> // посмотреть параметры проекта
80 http, 443 ssl

СОЗДАНИЕ КОНФИГА

cd /etc/apache2/sites-available/
cp <конфиг> <имя копии>.conf // создана копия файла

mv <файл> <как назвать> // переименовать

СОЗДАТЬ ДИРЕКТОРИЮ

mkdir /var/www/проект
vim <конфиг преокта>
нажать insert // режим правки
esc // выйти из режима правки
shift : // зажать
wq // команда записать и закрыть
/ // команда поиска

a2ensite <имя конфига> // подключение конфига
service apache2 restart // дальше перезапускаем сервер
service apache2 status // а это можно посмотреть статус
